## Overview

Adds smart recurring tasks, habits, and scheduled note creation to the vault.

## Files to Add

# Scheduler — Recurring Tasks & Habit Tracking

Always respond in the user's language.

## Core Capabilities

1. **Recurring Tasks**: Create notes on a schedule (daily, weekly, monthly)
2. **Habit Tracking**: Track completion streaks for habits
3. **Deadline Pre-warnings**: Create warning notes before deadlines
4. **Smart Scheduling**: Natural language scheduling ("every Monday", "first of month")
5. **Streak Reports**: Show habit completion over time

## Data Location

- Config: `Meta/recurring-tasks.yaml`
- State: `Meta/states/scheduler.md`
- Habits: `Meta/habits.md`

## Procedures

### Add Recurring Task

1. Ask user for:
   - Task name
   - Frequency (daily, weekly, monthly, custom)
   - Template to use (or create new)
   - Where to save notes

2. Add to `Meta/recurring-tasks.yaml`

3. Confirm with user

### List Recurring Tasks

Read `Meta/recurring-tasks.yaml` and display formatted list.

### Show Habit Streaks

Read `Meta/habits.md` and display streak information.

## Integration with Existing Skills

- Works with `/weekly-agenda` to include recurring tasks
- Works with `/deadline-radar` for pre-deadline warnings
- Triggers `/vault-audit` to clean up old recurring notes

## Output Format

- Next occurrence: Tomorrow at 07:00

### Suggested next agent

### 4. Background Scheduler Script

Location: `.claude/plugins/scheduler/check-schedule.py`

```python
#!/usr/bin/env python3
"""Check and execute scheduled recurring tasks."""

import yaml
import re
from pathlib import Path
from datetime import datetime, timedelta
from croniter import croniter

VAULT_ROOT = Path.cwd()
CONFIG_PATH = VAULT_ROOT / "Meta" / "recurring-tasks.yaml"
STATE_PATH = VAULT_ROOT / "Meta" / "states" / "scheduler.md"

def load_config():
    """Load recurring tasks configuration."""
    if not CONFIG_PATH.exists():
        return {"tasks": []}
    
    with open(CONFIG_PATH) as f:
        return yaml.safe_load(f) or {"tasks": []}

def should_run_task(task, last_check: datetime, now: datetime) -> bool:
    """Check if a task should run now."""
    if not task.get("enabled", True):
        return False
    
    freq = task.get("frequency")
    schedule = task.get("schedule", {})
    
    if freq == "daily":
        task_time = schedule.get("time", "00:00")
        hour, minute = map(int, task_time.split(":"))
        # Should run if we passed the task time since last check
        task_datetime = now.replace(hour=hour, minute=minute, second=0, microsecond=0)
        return last_check < task_datetime <= now
    
    elif freq == "weekly":
        day = schedule.get("day", "monday").lower()
        task_time = schedule.get("time", "00:00")
        days = ["monday", "tuesday", "wednesday", "thursday", "friday", "saturday", "sunday"]
        target_day = days.index(day)
        hour, minute = map(int, task_time.split(":"))
        
        # Check if today is the target day and we've passed the time
        if now.weekday() == target_day:
            task_datetime = now.replace(hour=hour, minute=minute, second=0, microsecond=0)
            return last_check < task_datetime <= now
        return False
    
    elif freq == "monthly":
        day = schedule.get("day", 1)
        task_time = schedule.get("time", "00:00")
        hour, minute = map(int, task_time.split(":"))
        
        if now.day == day:
            task_datetime = now.replace(hour=hour, minute=minute, second=0, microsecond=0)
            return last_check < task_datetime <= now
        return False
    
    elif freq == "custom" and "cron" in schedule:
        # Use cron expression
        itr = croniter(schedule["cron"], last_check)
        next_run = itr.get_next(datetime)
        return last_check < next_run <= now
    
    return False

def create_note_from_template(task, now: datetime) -> Path:
    """Create a note from template for a recurring task."""
    template_path = VAULT_ROOT / task["template"]
    location_template = task["location"]
    
    # Replace date placeholders
    location = location_template
    location = location.replace("{{date:YYYY-MM-DD}}", now.strftime("%Y-%m-%d"))
    location = location.replace("{{date}}", now.strftime("%Y-%m-%d"))
    
    note_path = VAULT_ROOT / location
    note_path.parent.mkdir(parents=True, exist_ok=True)
    
    # Read template
    if template_path.exists():
        content = template_path.read_text()
    else:
        # Create default template content
        content = f"""---
type: recurring
habit: {task.get("habit_tracking", False)}
created: "{now.isoformat()}"
task_id: {task["id"]}
---

# {task["name"]}

{task.get("description", "")}

## Status

- [ ] Completed

## Notes

"""
    
    # Add frontmatter date
    content = content.replace("{{date}}", now.strftime("%Y-%m-%d"))
    content = content.replace("{{time}}", now.strftime("%H:%M"))
    
    note_path.write_text(content)
    return note_path

def update_habit_tracking(task, now: datetime):
    """Update habit tracking for a recurring task."""
    if not task.get("habit_tracking"):
        return
    
    habits_path = VAULT_ROOT / "Meta" / "habits.md"
    
    # This is a simplified version - real implementation would parse and update
    # the habits.md file properly
    if not habits_path.exists():
        habits_path.parent.mkdir(parents=True, exist_ok=True)
        habits_path.write_text(f"""---
type: meta
agent: scheduler
---

# Habit Tracking

## {task["name"]}

- Started: {now.strftime("%Y-%m-%d")}
- Current streak: 0 days
- Last 7 days: ⏳⏳⏳⏳⏳⏳⏳
""")

def check_and_run_tasks():
    """Main scheduler check."""
    config = load_config()
    
    # Load last check time
    last_check = datetime.now() - timedelta(hours=1)  # Default to 1 hour ago
    if STATE_PATH.exists():
        content = STATE_PATH.read_text()
        # Parse last-check from state file
        for line in content.split("\n"):
            if line.startswith("last-check:"):
                try:
                    last_check = datetime.fromisoformat(line.split(":", 1)[1].strip())
                except:
                    pass
                break
    
    now = datetime.now()
    ran_tasks = []
    
    for task in config.get("tasks", []):
        if should_run_task(task, last_check, now):
            task_type = task.get("type", "note")
            
            if task_type == "note":
                note_path = create_note_from_template(task, now)
                update_habit_tracking(task, now)
                ran_tasks.append(f"Created: {note_path.name}")
            
            elif task_type == "skill":
                # Record that skill should be triggered
                ran_tasks.append(f"Skill: {task['skill']}")
            
            elif task_type == "deadline_warning":
                # Check for upcoming deadlines
                ran_tasks.append("Checked deadlines")
    
    # Update state
    STATE_PATH.parent.mkdir(parents=True, exist_ok=True)
    STATE_PATH.write_text(f"""---
agent: scheduler
---

## Post-it

last-check: {now.isoformat()}

### Recent runs
{chr(10).join(f"- {t}" for t in ran_tasks[-10:])}
""")
    
    if ran_tasks:
        print(f"Scheduler ran {len(ran_tasks)} task(s)")
        for t in ran_tasks:
            print(f"  - {t}")
    else:
        print("No tasks due")

if __name__ == "__main__":
    check_and_run_tasks()
```

### 7. Cron Setup for Background Execution

Add to user's crontab:

```bash
# Check recurring tasks every 15 minutes
*/15 * * * * cd /path/to/vault && python3 .claude/plugins/scheduler/check-schedule.py >> Meta/scheduler.log 2>&1
```

Or for systems without cron, add to the hooks system:

```bash
# .claude/hooks/post-command.sh - Run scheduler after each Claude interaction
python3 .claude/plugins/scheduler/check-schedule.py &
```

## Installation Steps

1. Copy `scheduler.md` to `agents/`
2. Create directory `.claude/plugins/scheduler/`
3. Copy Python script to that directory

5. Copy templates to `Templates/` (optional)
6. Add to `references/agents-registry.md`
7. Update `CLAUDE.md` dispatcher routing
8. Set up cron job or use hook-based execution
